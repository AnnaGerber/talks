var five = require("johnny-five"), servo, board;
board = new five.Board({port:'/dev/tty.usbserial-A900F86Y'});
board.on('ready', function () {
  // sound sensor
  var snd = new five.Sensor({pin: "A0", freq: 100});
  // tentacle / head servo
  hServo = five.Servo({
    pin: 10,
    range: [ 0, 180 ]
  });
  // tail servo
  tServo = five.Servo({
    pin: 9,
    range: [ 0, 120 ]
  });
  snd.on('change', function (err, reading) { 
    hServo.center();
  });
  // Add servos to REPL
  board.repl.inject({
    tail: tServo,
    head: hServo
  });
  hServo.sweep();
});
